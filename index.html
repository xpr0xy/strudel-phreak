strudel cheatsheet

ctrl+enter evaluate code

ctrl+/ toggle help

ctrl+. silence all

patterns:

note("c4 e4 g4 b4") — notes

s("bd sd hh cp") — samples

sound("gm_electric_guitar_clean") — general midi

freq(440) — frequency

patterns:

fast(2) / slow(2) — speed

rev — reverse

every(4, rev) — conditional

chunk(4, fast(2)) — subdivision

stack() — layer patterns

cat() — sequence

effects:

.lpf(800) — lowpass filter

.hpf(200) — highpass filter

.delay(0.5) — delay

.room(0.5) — reverb

.gain(0.8) — volume

.pan("<-1 1>") — panning
