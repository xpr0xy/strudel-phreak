strudel cheatsheet

ctrl+enter evaluate code

ctrl+/ toggle help

ctrl+. silence all

sounds:

s("bd sd hh cp") — drum samples

note("c4 e4 g4") — melodic notes

sound("gm_electric_piano") — general midi

patterns:

fast(2) / slow(2) — speed up/slow down

rev — reverse

every(4, rev) — apply every N cycles

stack() — layer patterns together

cat() — sequence patterns

effects:

.lpf(800) — lowpass filter

.hpf(200) — highpass filter

.delay(0.5) — delay

.room(0.5) — reverb

.gain(0.8) — volume

.pan("-1 1") — stereo pan
